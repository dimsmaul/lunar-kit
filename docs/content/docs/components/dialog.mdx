### Controlled State

<DynamicCodeBlock lang="tsx" code={`const [open, setOpen] = useState(false);

<Dialog open={open} onOpenChange={setOpen}>
  <DialogTrigger>
    <Button>Open Dialog</Button>
  </DialogTrigger>
  <DialogContent>
    <DialogHeader>
      <DialogTitle>Controlled Dialog</DialogTitle>
      <DialogDescription>
        This dialog's state is controlled externally.
      </DialogDescription>
    </DialogHeader>
    <DialogFooter>
      <Button onPress={() => setOpen(false)}>Close</Button>
    </DialogFooter>
  </DialogContent>
</Dialog>

{/* Open programmatically */}
<Button onPress={() => setOpen(true)}>
  Open from outside
</Button>`} />

### Confirmation Dialog

<DynamicCodeBlock lang="tsx" code={`export function ConfirmDialog({ onConfirm, onCancel }) {
  return (
    <Dialog>
      <DialogTrigger>
        <Button variant="destructive">Delete Account</Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Are you sure?</DialogTitle>
          <DialogDescription>
            This action cannot be undone. This will permanently delete your
            account and remove your data from our servers.
          </DialogDescription>
        </DialogHeader>
        <DialogFooter>
          <DialogClose>
            <Button variant="outline" onPress={onCancel}>
              Cancel
            </Button>
          </DialogClose>
          <Button variant="destructive" onPress={onConfirm}>
            Delete
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}`} />
